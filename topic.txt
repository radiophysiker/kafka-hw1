# Входим в контейнер
docker exec -it {container_id} bash

# Команда для создания топика:
kafka-topics.sh --create --bootstrap-server localhost:9094 --replication-factor 2 --partitions 3 --topic my-topic
// Created topic my-topic.

# Вывод подробной информации о топике my-topic:
kafka-topics.sh --describe --topic my-topic --bootstrap-server localhost:9094
